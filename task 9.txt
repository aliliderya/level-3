#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "Ededi daxil et: ";
    cin >> n;

    int say = 0;
    for (int i = 2; i <= n / 2; i++) {
        if (n % i == 0)
            say++;
    }

    if (say == 0)
        cout << "Sade ededdir.";
    else
        cout << "Murekkeb ededdir.";

    return 0;
}
