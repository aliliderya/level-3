#include <iostream>
using namespace std;

int main() {
    int n, ters = 0, original;
    cout << "Ededi daxil et: ";
    cin >> n;
    original = n;

    while (n > 0) {
        int reqem = n % 10;
        ters = ters * 10 + reqem;
        n = n / 10;
    }

    cout << "Tersi: " << ters << endl;

    if (original == ters)
        cout << "Palindrome ededdir." << endl;
    else
        cout << "Palindrome deyil." << endl;

    return 0;
}
